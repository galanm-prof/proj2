 spec COUNTRYCITY_SPORTSHOTEL_SERVICE
 Vocabulary:
    travel,
    portal,
    protont
 Inputs:
    travel:City _CITY,
    portal:Country _COUNTRY
 Outputs:
    travel:Sports _SPORTS,
    travel:Hotel _HOTEL
 Precondition :
    _CITY belongs locatedIn value _COUNTRY
 Effect:
    {_CITY} equivalentTo (hasActivity value _SPORTS),
    {_CITY} equivalentTo (hasAccommodation value _HOTEL)
